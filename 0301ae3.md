Jan03: Juggling Perl Versions

add\_perl5lib () {
  guts\_perl5lib ${1} add
}

del\_perl5lib ()
{
  guts\_perl5lib ${1} del
}

guts\_perl5lib ()
{
  for i in $(find ${1}/lib -name ${PERL\_CURRENT\_VERSION} -type d | \\
 sort -r)
  do
    ${2}path -f -p PERL5LIB $(dirname $i)
  done
  export PERL5LIB
}
